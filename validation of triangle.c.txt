#include <stdio.h>

int main() {
    float a, b, c;

    printf("Enter three sides of the triangle: ");
    scanf("%f %f %f", &a, &b, &c);
    if ((a + b > c) && (a + c > b) && (b + c > a)) {
        printf("The sides form a valid triangle.\n");
        if (a == b && b == c) {
            printf("Type: Equilateral Triangle (All sides are equal).\n");
        } 
        else if (a == b || b == c || a == c) {
            printf("Type: Isosceles Triangle (Two sides are equal).\n");
        } 
        else {
            printf("Type: Scalene Triangle (All sides are different).\n");
        }
    } 
    else {
        printf("The sides do not form a valid triangle.\n");
    }

    return 0;
}
