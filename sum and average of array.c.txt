
#include <stdio.h>

int main() {
  int n, i, arr[100]; 
  float sum = 0.0, average; 

  printf("Enter the number of elements (up to 100): ");
  scanf("%d", &n);

  printf("Enter %d numbers:\n", n);
  for (i = 0; i < n; i++) {
    scanf("%d", &arr[i]);
    sum += arr[i]; 
  }

  average = sum / n;

  printf("Sum = %.2f\n", sum);
  printf("Average = %.2f\n", average);

  return 0;
}
